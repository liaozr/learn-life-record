git git clone 你的项目地址.git git branch -r //查看远程分支 git branch //查看本地分支 git branch dev(分支名称, 可以随便取) //创建本地分支 git branch -d dev // 删除本地分支 git checkout dev // 切换分支 git push origin dev //发布本地分支到远程分支 git push --set-upstream origin dev // 建立本地分支与远程分支的关联 git add . // 把工作区更改的所有文件都添加进暂存区 git commit -m "说明信息" // 提交暂存区文件到本地仓库 git pull origin dev // 从远程某个分支拉取代码 git push --delete origin dev(分支名称) // 删除远程仓库分支 //合并分支 # 切换到稳定的目标分支 git checkout master # 更新最新代码，防止本地仓库对应分支代码不够新而出现问题 git pull # 合并分支到本地仓库 git merge dev # 发布合并后的新代码到远程分支 git push
